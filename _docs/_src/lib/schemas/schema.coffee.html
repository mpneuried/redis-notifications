36    validate: ( data )=>
37      for _k, def of @schema
38        _val = data[ _k ]
39        if def.required and not _val?

41        else if not _val?
42          if def.default?
43            if _.isFunction( def.default )
44              _val = data[ _k ] = def.default( data, def )
45            else
46              _val = data[ _k ] = def.default

48        switch def.type

50            if not _.isObject( _val ) and def instanceof Schema

56            if not _.isNumber( _val )

60            if not _.isArray( _val )

62  
63          when "boolean"
64            if not _.isBoolean( _val )

66  
67          when "object"

79          when "timezone"

84          return @_error( "regexp", _k, def, { regexp: regexp.toString() } )

86        if def.type is "string" and def.sanitize
